Configuración inicial

Necesitás conectar la app a una hoja de Google Sheets. Solo se hace una vez.

Pasos para configurar
1
Abrí Google Sheets y creá una hoja nueva llamada Inventario
2
Hacé clic en Extensiones → Apps Script en el menú superior
3
Borrá el código que hay y pegá este:
function doPost(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName("Inventario") || ss.getActiveSheet(); // Agregar encabezados si la hoja está vacía if (sheet.getLastRow() === 0) { sheet.appendRow(["Fecha/Hora","Código","Cantidad","Ubicación","Vencimiento","Notas"]); sheet.getRange(1,1,1,6).setFontWeight("bold").setBackground("#1a73e8").setFontColor("#ffffff"); sheet.setFrozenRows(1); } var data = JSON.parse(e.postData.contents); sheet.appendRow([ data.fecha, data.codigo, data.cantidad, data.ubicacion, data.vencimiento, data.notas || "" ]); return ContentService .createTextOutput(JSON.stringify({status:"ok", row: sheet.getLastRow()})) .setMimeType(ContentService.MimeType.JSON); } function doGet(e) { return ContentService .createTextOutput(JSON.stringify({status:"ok"})) .setMimeType(ContentService.MimeType.JSON); }
4
Hacé clic en Implementar → Nueva implementación, elegí tipo Aplicación web, y en "Quién tiene acceso" seleccioná Cualquier usuario. Hacé clic en Implementar y copiá la URL que aparece.
5
Pegá la URL acá abajo:
